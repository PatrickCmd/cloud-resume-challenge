#
# Installation:
#   pip install -r requirements.txt
#
# Installation in virtual environment (recommended):
#   python3 -m venv venv
#   source venv/bin/activate  # On Windows: venv\Scripts\activate
#   pip install -r requirements.txt

#   pip install -r requirements.txt --upgrade

# Ansible
# Automation platform for configuration management and deployment
# Version 8.0+ required for modern callback plugins and features

# Boto3
# AWS SDK for Python - required by all Ansible AWS modules
# Provides programmatic access to AWS services

# Must be compatible with boto3 version

# Usually installed with Ansible, but specified for completeness

# Required for parsing Ansible playbooks and variables

# Provides secure cryptographic recipes

# Packaging

# Used by Ansible for version comparisons
